def prime(a,b):
    '''
    Returns primes in the interval [a,b] in a list
    '''
    L=[]
    for num in range(a,b+1):
    c=0
    for i in range(1,int(np.sqrt(b))+1):
        if (num % i==0):
            c=c+1
    if (c<=1):
        L.append(num)
    return L

class complex():
    def __init__(self,real_part,imaginary_part):
    self.real=real_part
    self.imaginary=imaginary_part

    def __str__(self):
        if self.imaginary < 0:
            return f'{self.real}{self.imaginary}i'
        if self.imaginary > 0:
            return f'{self.real}+{self.imaginary}i'
        if self.imaginary==0:
            return f'{self.real}'

    def conjugate(self):
    '''
    returns complex conjugate
    '''
    return complex(self.real,-self.imaginary)

    def absolute_value(self):
    '''
    returns absolute value
    '''
    return np.sqrt((self.real)**2+(self.imaginary)**2)

    def addition(self,number):
    '''
    perfor addition
    '''
    return complex(self.real+number.real,self.imaginary+number.imaginary)

    def substraction(self,number):
    '''
    perform substraction
    '''
    return complex(self.real-number.real,self.imaginary-number.imaginary)

    def multiply(self,number):
    '''
    perform multiplication
    '''
    return complex(self.real*number.real-self.imaginary*number.imaginary,self.real*number.imaginary+self.imaginary*number.real)

    def divide(self,number):
    '''
    perform divsion
    '''
    result_real=(self.real*number.real+self.imaginary*number.imaginary)/((number.real)**2+(number.imaginary)**2)
    result_imag=(self.imaginary*number.real-self.real*number.imaginary)/((number.real)**2+(number.imaginary)**2)
    result=complex(result_real,result_imag)
    return result

    def angle(self):
    '''
    returns angle between -pi/2 and pi/2
    '''
    return np.arctan(self.imaginary/self.real)

mu_x=5

Random points inside the ellipse

Equation of an ellipse with center $(\mu_{x},\mu_{y})$ with major axis $2a=10$ and minor axis $2b=5$ can be parameterized by the equations

$x=\mu_{x}+a*cos(t), y=\mu_{y}+b*sin(t)$, $t \in [0,2\pi]$

To generate a point inside this ellipse, take random points $a_{1}$,$b_{1}$,$t$ in the interval $[0,a]$, $[0,b]$, $[0,2\pi]$ respectively, then the point

$(x,y)$ where $x=\mu_{x}+a_{1}*cos(t)$, $y=\mu_{y}+b_{1}*sin(t)$

is a point inside the ellipse.

Note: The equations

$x=\mu_{x}+a_{1}*cos(t)$, $y=\mu_{y}+b_{1}*sin(t)$, $t \in [0,2\pi]$

represents an ellipse which lies inside the ellipse

$x=\mu_{x}+a*cos(t), y=\mu_{y}+b*sin(t)$, $t \in [0,2\pi]$

if $a_{1}<a$, ${b_1} < b$. Fixing a value of $t \in [0,2\pi]$, will give a random point inside the ellipse.
