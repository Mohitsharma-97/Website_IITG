mycholb
function [G] = mycholb(A)
% Cholesky decomposition of A is G^T*G .
n=length(A(:,1));
G=zeros(n,n);
if A(1,1)<0
exit('Matrix not positive definite')
end
G(1,1)=sqrt(A(1,1));
for j=2:n
h=linsolve(G(1:j-1,1:j-1)',A(1:j-1,j));
G(1:j-1,j)=h;
if A(j,j)-norm(h)^2 < 0
exit('Matrix not positive definite')
end
G(j,j)=sqrt(A(j,j)-norm(h)^2);
end
end

function [Q,R] = cgs(V)
% orthonormalize the columns of an n × m matrix V; (n ≥ m) by the Classical
% Gram Schmidt procedure so that Q is an isometry whose columns
% are obtained by applying Classical Gram-Schimdt Procedure to columns of V
% and R is an upper traingular matrix such that R(i,j)=<V(:,j),Q(:,i)>.
[n,m]=size(V);
Q=zeros(n,m);
Q(:,1)=V(:,1)./norm(V(:,1)); % Orthonormalization of first column of V
R=zeros(m,m);
R(1,1)=norm(V(:,1));
for j=2:m
coeff=V(:,j)'*Q(:,1:j-1); % column vector which stores entries such that
% q_j=v_j-coeff(1)*q_1-coeff(2)*q_2-...-coeff(j-1)*q_(j-1)
q_j=V(:,j)-sum(coeff.*Q(:,1:j-1),2);
norm_q_j=norm(q_j);
if norm_q_j==0
exit('Columns of given matrix are dependent')
end
Q(:,j)=q_j./norm_q_j; %jth orthonormalized vector
R(1:j-1,j)=coeff'; % Store the coeff vector as the jth column in R
R(j,j)=norm_q_j; % (j,j) entry of R is ||q_j||
end
 
end

function [Q,R] = mgs(V)
% orthonormalize the columns of an n × m matrix V; (n ≥ m) by the Classical
% Gram Schmidt procedure so that Q is an isometry whose columns
% are obtained by applying Classical Gram-Schimdt Procedure to columns of V
% and R is an upper traingular matrix such that R(i,j)=<V(:,j),Q(:,i)>.
[n,m]=size(V);
Q=V;
Q(:,1)=V(:,1)./norm(V(:,1)); % Orthonormalization of first column of V
R=zeros(m,m);
R(1,1)=norm(V(:,1));
for j=2:m
for k=1:j-1
R(k,j)=Q(:,j)'*Q(:,k);
Q(:,j)=Q(:,j)-R(k,j).*Q(:,k);
end
R(j,j)=norm(Q(:,j));
if R(j,j)==0
exit('Columns of given matrix are dependent')
end
Q(:,j)=Q(:,j)./norm(Q(:,j)); %jth orthonormalized vector
end
 
end

function [Q,R] = cgsrep(V)
% orthonormalize the columns of an n × m matrix V; (n ≥ m) by the Classical
% Gram Schmidt procedure so that Q is an isometry whose columns
% are obtained by applying Classical Gram-Schimdt Procedure to columns of V
% and R is an upper traingular matrix such that R(i,j)=<V(:,j),Q(:,i)>.
[n,m]=size(V);
Q=V;
Q(:,1)=V(:,1)./norm(V(:,1));% Orthonormalization of first column of V
R=zeros(m,m);
R(1,1)=norm(V(:,1));
for j=2:m
for i=1:j-1
R(i,j)=Q(:,j)'*Q(:,i);
Q(:,j)=Q(:,j)-R(i,j).*Q(:,i);
end
for i=1:j-1
a=Q(:,j)'*Q(:,i);
Q(:,j)=Q(:,j)-a.*Q(:,i);
R(i,j)=R(i,j)+a;
end
R(j,j)=norm(Q(:,j));
if R(j,j)==0
exit('Columns of the input matrix are linearly independent')
end
Q(:,j)=Q(:,j)./R(j,j);
end
end
