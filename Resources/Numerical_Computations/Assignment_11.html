reflect
function [u,gamma,tao]=reflect(x)
% x is column vector
% Determine reflector Q such that Qx=[-tao,0,0,...,0]
% and Q = I_n - gamma*u*u^T
% u is column vector
n=length(x);
tao=sign(x(1))*norm(x);
y=zeros(n,1);
y(1)=-tao;
u=(x-y)./(tao+x(1));
gamma=(tao+x(1))/tao;
 
end
applreflect
function [B] = applreflect(u,gamma,A)
% Compute B=Q*A where Q is the reflector I_n-gamma*u*u^T
v=gamma.*u;
w=u'*A;
C=v*w;
B=A-C;
end

function [Q,R] = reflectqr(A)
% Condensed QR decomposition A using reflectors.
% if A is nxm matrix then Q is nxm matrix and V is mxm
[n,m]=size(A);
Q=zeros(n,m);
if n>m
p=m;
elseif n==m
p=m-1;
else
exit('Number of columns are more than the number of rows.')
end
gammaVec=zeros(p,1); % strore gamma in gammaVec
for i=1:p
if norm(A(i+1:n,i))~=0
x=A(i:n,i);
[u,gamma,tao]=reflect(x);
gammaVec(i)=gamma;
A(i:n,i)=u;
A(i:n,i+1:m)=applreflect(u,gamma,A(i:n,i+1:m));
A(i,i)=-tao;
end
end
R=triu(A);
R=R(1:m,:);
U=tril(A,-1)+eye(n,m);
I=eye(n,m);
for k=1:p
x=I(k:n,k);
for j=k:-1:1
gamma=gammaVec(j);
u=U(j:n,j);
x=applreflect(u,gamma,x);
x=[0;x];
end
Q(:,k)=x(2:n+1);
end
if p==m-1
x=I(m:n,m);
x=applreflect(1,0,x);
x=[0;x];
for j=p:-1:1
gamma=gammaVec(j);
u=U(j:n,j);
x=applreflect(u,gamma,x);
x=[0;x];
end
Q(:,m)=x(2:n+1);
end
end
mycholb
function [G] = mycholb(A)
% Cholesky decomposition of A is G^T*G .
n=length(A(:,1));
G=zeros(n,n);
G(1,1)=sqrt(A(1,1));
for j=2:n
h=linsolve(G(1:j-1,1:j-1)',A(1:j-1,j));
G(1:j-1,j)=h;
G(j,j)=sqrt(A(j,j)-norm(h)^2);
end
end
 
 
rowforward
function x = rowforward(L,b)
%
n=length(b);
for k=1:n
for j=1:k-1
b(k)=b(k)-L(k,j)*b(j);
end
if L(k,k)~=0
b(k)=b(k)/L(k,k);
else
exit('matrix is singular')
end
end
x=b';
end
 
 
colbackward
function [x] = colbackward(U,b)
%
n=length(b);
for j=n:-1:1
if U(j,j)~=0
b(j)=b(j)/U(j,j);
end
for i=1:j-1
b(i)=b(i)-U(i,j)*b(j);
end
end
x=b';
end
Horner
function y = Horner(p,x)
% Evaluates a polynomial p(z)=p_1*z^n+p_2*z^(n-1)+...+p_n*z+p_(n+1)
% at z=x by taking x in real numbers and a vector p with p(i)=p_i, i=1:n+1
% as input.
y=0;
for i=1:(length(p)-1)
y=x.*(p(i)+y);
end
y=p(length(p))+y;
end
