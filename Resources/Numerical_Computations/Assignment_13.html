function [iter,lambda] = Powermethod(A,x,k)
% Input: nxn complex matrix A, initial vector x in C^n, k= no of iterations
% Output : nxk matrix iter whose jth column is the iterate jth iterate q_j
% and a scalar lambda which is dominant eigenvalue.
[~,r]=max(abs(x));
s=x(r);
iter=[];
q=x/s;
for j=1:k
q=A*q;
[~,r]=max(abs(q));
s=q(r);
q=q/s;
iter=[iter,q];
end
lambda=s;
end

p=p';
for k=1:n-1
[r,m]=max(abs(A(k:n,k)));
m=m+k-1;
if m~=k
A([k,m],:)=A([m,k],:);
p([k,m])=p([m,k]);
end
if A(k,k)~=0
A(k+1:n,k)=A(k+1:n,k)/A(k,k);
end
A(k+1:n,k+1:n)=A(k+1:n,k+1:n)-A(k+1:n,k)*A(k,k+1:n);
end
L=eye(n)+tril(A,-1);
U=triu(A);
end

rowforward

function x = rowforward(L,b)
%
n=length(b);
for k=1:n
for j=1:k-1
b(k)=b(k)-L(k,j)*b(j);
end
if L(k,k)~=0
b(k)=b(k)/L(k,k);
else
exit('matrix is singular')
end
end
x=b';
end
 

colbackward

function [x] = colbackward(U,b)
%
n=length(b);
for j=n:-1:1
if U(j,j)~=0
b(j)=b(j)/U(j,j);
end
for i=1:j-1
b(i)=b(i)-U(i,j)*b(j);
end
end
x=b';
end

Shiftinv

function [iter,lambda]=Shiftinv(A,x,s,k)
% Performs k iterations of shift and inverse method using an nxn complex
% matrix A, x in C^n, and shift s and returns an nxk matrix iter whose jth
% column is the jth iterate q_j and a scalar lamda is the eigenvalue of A
% closest to s.
[L,U,p]=gepp(A-s*eye(length(A)));
[~,r]=max(abs(x));
lambda=x(r);
iter=[];
q=x/lambda;
for j=1:k
b=q(p);
y=rowforward(L,b);
q=colbackward(U,y);
[~,r]=max(abs(q));
lambda=q(r);
q=q/lambda;
iter=[iter,q];
end
lambda=(1/lambda)+s;
end

Rayleigh

function [iter,lambda] = Rayleigh(A,x,k)
% Performs k iterations of inverse iterations with Rayleigh quotient shifts
% using A, an nxn complex matrix, and x in C^n.
% Returns an nxk matrix iter whose jth column is the jth iterate qj and a
% scaler lambda is the eigenvalue of A to which the Rayleigh quotient
% shifts converge.
[~,r]=max(abs(x));
s=x(r);
iter=[];
q=x/s;
[Q,H]=hess(A);
rayquo=((q')*H*q)/((q')*q);
for i=1:k
q=(H-rayquo*eye(length(A)))\q;
[~,r]=max(abs(q));
s=q(r);
q=q/s;
rayquo=((q')*H*q)/(q'*q);
iter(:,i)=Q*q;
end
lambda=(1/s)+rayquo;
end
