Newton function

function [y,n] = newton(f,g,x,N,tol)
% The function performs Newton operations to find a column vector y of
% length N such that |f(y(N))| < tol with initial guess x. The f and g
% should be the function handles of f and its first derivative.
iter=0;
y=[x];
while(abs(f(x))>tol) && (iter<N)
x=x-f(x)/g(x);
y=[y x];
iter=iter+1;
if (iter==N) && (abs(f(y(N+1)))>=tol)
disp('zero could not be found for the given tolerance within the given iterations')
end
end
y=y';
n=iter;
end

Regula_falsi function

function [y,n] = regula_falsi(f,a,b,tol)
%Finds a root of a function f in the interval [a,b](where f(a)f(b) < 0)
% up to a given tolerance tol using the reugla_falsi method.
iter=0;
m=a;
y=m;
while(abs(m-2^(1/4)) > tol)
iter=iter+1;
m=a-(b-a)/((f(a)/f(b))-1);
if sign(f(a))==sign(f(m))
a=m;
else
b=m;
end
y=[y m];
end
y=y';
n = iter;
end
 

Secant function

function [y,n] = secant(f,x_0,x_1,tol)
% Uses secant method to calculate the root of a function f, with inital
% guess x_0 and x_1 until subsequent iterations are within the tolerance
% tol. It returns the root y and integer n, where y=x_n such that
% |x_n-x_(n-1)| < tol.
a=x_0;
b=x_1;
c=0;
iter=0;
y=[a b];
while(abs(b-a)>tol)
c=b-((b-a)/(f(b)-f(a)))*f(b);
a=b;
b=c;
iter=iter+1;
y=[y c];
end
n=iter;
y=y';
end
 

Fixed point iteration function

function [y,n]=fixedpoint(f,x_0,acc,N)
% Performs fixed point iterations on the function f with starting point
% x_0 until the desired accuracy acc is achieved or the number of iterations N is exhausted.
a=x_0;
iter=0;
y=a;
while (abs(f(a)-a)>acc) && (iter < N)
a=f(a);
iter=iter+1;
y=[y a];
end
if (iter==N) && (abs(f(a)-a)>acc)
disp(['fixed point could not be found for the given accuracy within the given iterations.' ...
' Have a look at the iterations.']);
end
y=y';
n=iter;
end

function [y,n] = newton_sqrt(a,x_0,tol)
% Calculates the square root of a number 'a' using Newton-Raphson Method,
% starting with an initial guess x_0 > 0 and iterations run until desired
% accuracy tol is achieved. It returns the computed value y in n
% iterations, where n is the iteration number after which desire accuracy
% was achieved.
 
x=x_0;
y=[x_0];
n=0;
while abs(x-sqrt(a)) > tol
x=(1/2)*(x+a/x);
n=n+1;
y=[y x];
 
 
end
y=y';
end

function [y,T] = newtonS(f,g,x,N,tol)
%find an approximate solution y = (y(1),y(2)) of a nonlinear
% system of equations f(u,v) = 0 via Newton's method.
% The inputs should be the function handles of f and its Jacobian matrix g,
% initial guess x, maximum number of iterations N and tolerance tol such that
% norm(f(y(1),y(2))) < tol. Also returns T, the number of iterations
% required to achieve norm(f(y(1),y(2))) < tol.
iter=0;
while (norm(f(x(1),x(2)))>tol) && (iter < N)
s=linsolve(g(x(1),x(2)),-f(x(1),x(2)));
x=x+s';
iter=iter+1;
T=iter;
end
y=x;
if (iter==N) && (norm(f(x(1),x(2)))>tol)
error('zero could not be found for the given tolerance within the given iterations');
end
end
