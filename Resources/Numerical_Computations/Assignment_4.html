polyinterp

function [v] = polyinterp(x,f,u)
% Construct Lagrange interpolant p_n(x)=sum(l_j(x)*f_j) of the data
% (x_1,f_1),...,(x_n,f_n) and evaluates p_n(x) at the points given in the
% vector u, so v(i)=p_n(u(i)). x,f,u and v are all column vectors.
l_denom=[];
for i=1:length(x)
li=1;
for j=1:length(x)
if j~=i
li=li*(x(i)-x(j));
end
end
l_denom=[l_denom;li];
end
 
for k=1:length(u)
l_num=[];
for j=1:length(x)
l_num_entry=1;
for i=1:length(x)
if i~=j
l_num_entry=l_num_entry*(u(k)-x(i));
end
end
l_num=[l_num;l_num_entry];
end
l=l_num./l_denom;
v(k)=sum(l.*f);
 
end
v=v';
end

barycent

function [v] = barycent(x,f,u)
% Generates the Lagrange interpolant p_n(x) in barycentric form
% and computes v(i)=p_n(u(i)). x,f,u and v all must be column vectors.
w=[]; % w vector will conatin all w_j's
for j=1:length(x)
wj=1;
for i=1:length(x)
if i~=j
wj=wj*(x(j)-x(i));
end
end
wj=1./wj;
w=[w;wj];
end
for k=1:length(u)
[r,p]=min(abs(u(k)-x));
if r==0
v(k)=f(p);
else
factor=w./(u(k)-x);
numerator=sum(factor.*f);
denominator=sum(factor);
v(k)=numerator/denominator;
end
end
v=v';
 
end

newtondd

function [c] = newtondd(x,f)
% Generates the vector of coefficients c of the Newton interpolating
% polynomial that passes through (x1, f1), . . . ,(xn, fn). Enter x and f
% as column vectors.The form of c is:
% c=[f[x1], f[x2,x1], f[x3,x2,x1],..., f[xn,...,x1]]
dd=zeros([length(x),length(x)]);
dd(:,1)=f;
for i=2:length(x)
for j=2:i
temp=(dd(i,j-1)-dd(i-1,j-1))/(x(i)-x(i-j+1));
dd(i,j)=temp;
end
end
for i=1:length(x)
c(i)=dd(i,i);
end
end

nest

function [y] = nest(c,x,u)
%takes the coefficient vector c from c= newtdd(x, f) and
% performs nested additions and multiplications to evaluate Newton
% interpolating polynomial p_n(x) that passes through (x1,f1),...(x_n,f_n)
% at entries of the vector u such that v=p_n(u)
y=c(1)*ones(size(u));
for i=2:length(c)
term=c(i);
for j=1:i-1
term=term.*(u-x(j));
end
y=y+term;
end
end

chebnodes function

function [x] = chebnodes(a,b,n)
% Generate n Chebyshev nodes in the interval [a,b]
j=1:n;
x=((b+a)/2)+((b-a)/2)*cos(((2*j-1)*pi)/(2*n));
end
