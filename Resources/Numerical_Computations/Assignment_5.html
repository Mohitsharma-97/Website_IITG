Trapezoid

function [y] = trapezoid(f,a,b,n)
% Evaluate integration of f(x) over the interval [a,b] via composite Trapezoid rule with n panels on [a,b].
h=(b-a)/n;
x=linspace(a,b,n+1);
f_at_x=f(x);
 
% integral_of_f_at_panels=zeros([1,n]);
% for j=1:n
% integral_over_xj_to_xj_plus_one=(h/2)*(f_at_x(j)+f_at_x(j+1));
% integral_of_f_at_panels(j)=integral_over_xj_to_xj_plus_one;
% end
weights=h*ones([1,n+1]);
weights(1)=h/2;weights(n+1)=h/2;
y=weights*f_at_x';
% y=sum(integral_of_f_at_panels);
end
 

Midpoint

function [y] = midpoint(f,a,b,n)
% Evaluate integration of f(x) over the interval [a,b] via composite midpoint rule with n panels on [a,b].
h=(b-a)/n;
x=linspace(a,b,n+1);
w=zeros([1,n]);
for j=1:n
w(j)=(x(j)+x(j+1))/2;
end
integral_of_f=h*sum(f(w));
y=integral_of_f;
 
end
 

Simpson

function [y] = simpsononethird(f,a,b,n)
% Evaluate integration of f(x) over the interval [a,b] via composite Simpson one-third rule with n panels on [a,b].
h=(b-a)/(2*n);
x=linspace(a,b,2*n+1);
f_at_x=f(x);
 
y=(h/3)*(f_at_x(1)+f_at_x(2*n+1)+2*sum(f_at_x(3:2:2*n-1))+4*sum(f_at_x(2:2:2*n)));
end
