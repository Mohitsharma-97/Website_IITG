function [w] = gaussweightslegendre( n )
% Generate n+1 weights for Gauss quadrature rule in the interval [-1,1]
% from the roots of Legendre polynomial.
syms x
roots=vpasolve(legendreP(n+1,x)==0);
% roots=double(roots); % comment this line if you want better accuracy or use it in the code if you want faster computation.
A=ones(size(roots'));
for i=1:n
A=[A;(roots').^i];
end
b=[];
for i=1:n+1
b=[b;((1)^i-(-1)^i)/i];
end
w=A\b;
end

function [y] = gaussquad( f,a,b,n )
% Computes integral of f over [a,b] via n-point Gauss Quadrature rule
% with n+1 Legendre nodes.
g=@(x)f((b+a)./2+((b-a)./2).*x).*((b-a)./2);
syms t;
roots=vpasolve(legendreP(n+1,t)==0);
w=gaussweightslegendre(n);
y=g(roots)'*w;
end

function [int,k] = adaptiveT(f,a,b,tol)
% Implements adaptive trapezoid rule for approximating the integral of f
% over the interal [a,b], where tol is the amount of error you are willing
% to tolerate in the approximation. The ouput int should record the
% computed value of the integral and the output k should be the number of
% sub-intervals of [a,b] created in the process.
 
int=0;
n=1;
a(1)=a;
b(1)=b;
tol(1)=tol;
app(1)=trap(f,a,b);
k=2; % Initializing k=2 beacause the interval [a,b] is divided into two subintervals immediately after entering into the while loop present in the next line.
while n>0 % n is current position at end of the list
c=(a(n)+b(n))/2;
oldapp=app(n);
app(n)=trap(f,a(n),c);
app(n+1)=trap(f,c,b(n));
if abs(oldapp-(app(n)+app(n+1)))<3*tol(n)
int=int+app(n)+app(n+1); % success
n=n-1; % done with interval
else % divide into two intervals
b(n+1)=b(n); b(n)=c; % set up new intervals
a(n+1)=c;
tol(n)=tol(n)/2; tol(n+1)=tol(n);
n=n+1; % go to end of list, repeat
k=k+1;
end
end
function s=trap(f,a,b)
s=(f(a)+f(b))*(b-a)/2;
 
end
 
end
