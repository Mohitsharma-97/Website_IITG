Eulerforward
function [x,y] = Eulerforward(f,int,ics,m)
% solve IVP dy/dx=f(x,y).f(u)=v f:[a,b]->R using Euler forward
% input: f is function handle
% int is interval [a.b]
% ics is initial conditions ics=[u v]
% m is number of steps
% OUtput: x is a vector of equidsitant points such that x[1]=u and x[m]=b
% y is vector such that y(i) is approximate value of y(x(i))
x=linspace(ics(1),int(2),m)';
h=x(2)-x(1);
y=[ics(2)];
for i=2:m
y(i)=y(i-1)+h*f(x(i-1),y(i-1));
end
y=y';
end

function [x,y] = RungeKuttasecond(f,int,ics,m)
% solve IVP dy/dx=f(x,y).f(u)=v f:[a,b]->R using Runge-Kutta Second second
% method
% input: f is function handle
% int is interval [a.b]
% ics is initial conditions ics=[u v]
% m is number of steps
% OUtput: x is a vector of equidsitant points such that x[1]=u and x[m]=b
% y is vector such that y(i) is approximate value of y(x(i))
x=linspace(ics(1),int(2),m)';
h=x(2)-x(1);
y=[ics(2)];
for i=1:m-1
y1=y(i);
y2=y(i)+(h/2)*f(x(i),y1);
y(i+1)=y(i)+h*f(x(i)+(h/2),y2);
end
y=y';
end

function [x,y] = RungeKuttafourth(f,int,ics,m)
% solve IVP dy/dx=f(x,y).f(u)=v f:[a,b]->R using Runge-Kutta Second fourth
% method
% input: f is function handle
% int is interval [a.b]
% ics is initial conditions ics=[u v]
% m is number of steps
% OUtput: x is a vector of equidsitant points such that x[1]=u and x[m]=b
% y is vector such that y(i) is approximate value of y(x(i))
x=linspace(ics(1),int(2),m)';
h=x(2)-x(1);
y=[ics(2)];
for i=1:m-1
y1=y(i);
y2=y(i)+(h/2)*f(x(i),y1);
y3=y(i)+(h/2)*f(x(i)+(h/2),y2);
y4=y(i)+h*f(x(i)+(h/2),y3);
y(i+1)=y(i)+(h/6)*(f(x(i),y1)+2*f(x(i)+(h/2),y2)+2*f(x(i)+(h/2),y3)+f(x(i+1),y4));
end
y=y';
end

function [x,Y] = Eulerforwardvec(f,int,ics,m)
%
x=linspace(ics(1),int(2),m)';
h=x(2)-x(1);
Y=[ics(2);ics(3)];
for i=1:m-1
Y(:,i+1)=Y(:,i)+h*f(x(i),Y(:,i));
end
