function A = Wilkinson(n)
%Generate Wilkinson matrix of size n.
A=-1*ones(n);
A=tril(A,-1);
A=A+eye(n);
A(:,n)=ones([n,1]);
 
end

function H = real_hamiltonian(n)

B=B+B1;

function x = rowforward(L,b)
%
n=length(b);
for k=1:n
for j=1:k-1
b(k)=b(k)-L(k,j)*b(j);
end
if L(k,k)~=0
b(k)=b(k)/L(k,k);
else
exit('matrix is singular')
end
end
x=b';
end

function [x] = colbackward(U,b)
%
n=length(b);
for j=n:-1:1
if U(j,j)~=0
b(j)=b(j)/U(j,j);
else
exit('matrix is singular')
end
for i=1:j-1
b(i)=b(i)-U(i,j)*b(j);
end
end
x=b';
end

function [L,U] = genp(A)
% LU decomposition of the input matrix A using
% Gaussian elimination with no pivoting.
n=length(A(1,:));
for k=1:n-1
if A(k,k)~=0
A(k+1:n,k)=A(k+1:n,k)/A(k,k);
else
exit('zero pivot encountered')
end
A(k+1:n,k+1:n)=A(k+1:n,k+1:n)-A(k+1:n,k)*A(k,k+1:n);
L=eye(n)+tril(A,-1);
U=triu(A);
end
end
