Functions

genp
function [L,U] = genp(A)
% LU decomposition of the input matrix A using
% Gaussian elimination with no pivoting.
n=length(A(1,:));
for k=1:n-1
if A(k,k)~=0
A(k+1:n,k)=A(k+1:n,k)/A(k,k);
else
exit('zero pivot encountered')
end
A(k+1:n,k+1:n)=A(k+1:n,k+1:n)-A(k+1:n,k)*A(k,k+1:n);
end
L=eye(n)+tril(A,-1);
U=triu(A);
end
rowforward
function x = rowforward(L,b)
%
n=length(b);
for k=1:n
for j=1:k-1
b(k)=b(k)-L(k,j)*b(j);
end
if L(k,k)~=0
b(k)=b(k)/L(k,k);
else
exit('matrix is singular')
end
end
x=b';
end
colbackward
function [x] = colbackward(U,b)
%
n=length(b);
for j=n:-1:1
if U(j,j)~=0
b(j)=b(j)/U(j,j);
else
exit('matrix is singular') %% exit with error message if U(j,j) = 0
end
for i=1:j-1
b(i)=b(i)-U(i,j)*b(j);
end
end
x=b';
end
gepp
function [L,U,p] = gepp(A)
% LU decomposition of the input matrix A using
% Gaussian elimination with partial pivoting.
n=length(A(1,:));
p=1:n;
p=p';
for k=1:n-1
[r,m]=max(abs(A(k:n,k)));
m=m+k-1;
if m~=k
A([k,m],:)=A([m,k],:);
p([k,m])=p([m,k]);
end
if A(k,k)~=0
A(k+1:n,k)=A(k+1:n,k)/A(k,k);
else
exit('matrix is singular')
end
A(k+1:n,k+1:n)=A(k+1:n,k+1:n)-A(k+1:n,k)*A(k,k+1:n);
end
L=eye(n)+tril(A,-1);
U=triu(A);
end
geppsolve
function [x]=geppsolve(A,b)
%solve Ax=b using gaussian eliination with partial pivoting
[L,U,p]=gepp(A);
b=b(p);
y=rowforward(L,b);
x=colbackward(U,y);
end
mydet
function [d]=mydet(A)
% Calculate determinant of the input matrix A using
% Gaussian elimination with partial pivoting.
n=length(A(1,:));
p=1:n;
p=p';
sign=1;
for k=1:n-1
[r,m]=max(abs(A(k:n,k)));
m=m+k-1;
if m~=k
A([k,m],:)=A([m,k],:);
p([k,m])=p([m,k]);
sign=sign*(-1);
end
if A(k,k)~=0
A(k+1:n,k)=A(k+1:n,k)/A(k,k);
end
A(k+1:n,k+1:n)=A(k+1:n,k+1:n)-A(k+1:n,k)*A(k,k+1:n);
end
d=sign*prod(diag(A));
 
end
